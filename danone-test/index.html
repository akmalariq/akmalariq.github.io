Web Scraping

Fetch All Laptops to CSV

Fetch all laptops, prices, and specifications from webscraper.io/test-sites/e-commerce/static/computers/laptops — no shortcuts — and store the result as a .csv file. The scraper auto-detects the total page count and crawls every page.

task1_scraper.py
task1_scraper.py 
import csv
import requests
from bs4 import BeautifulSoup

BASE_URL = (
    "https://webscraper.io/test-sites"
    "/e-commerce/static/computers/laptops"
)


def scrape_page(url: str) -> list[dict]:
    response = requests.get(url)
    response.raise_for_status()
    soup = BeautifulSoup(response.text, "html.parser")

    laptops = []
    for item in soup.select(".thumbnail"):
        name  = item.select_one(".title")
        price = item.select_one("[itemprop='price']")
        spec  = item.select_one(".description")

        laptops.append({
            "name": name.get_text(strip=True) if name else "",
            "price": (
                price.get_text(strip=True).replace("$", "")
                if price else ""
            ),
            "specification": spec.get_text(strip=True) if spec else "",
        })
    return laptops


def get_total_pages(url: str) -> int:
    soup = BeautifulSoup(requests.get(url).text, "html.parser")
    links = soup.select(".page-item a.page-link")
    numbers = [
        int(l.get_text(strip=True))
        for l in links
        if l.get_text(strip=True).isdigit()
    ]
    return max(numbers) if numbers else 1


def main():
    total_pages = get_total_pages(BASE_URL)  # → 20
    all_laptops = []

    for page in range(1, total_pages + 1):
        url = BASE_URL if page == 1 else f"{BASE_URL}?page={page}"
        all_laptops.extend(scrape_page(url))

    with open("laptops.csv", "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(
            f, fieldnames=["name", "price", "specification"]
        )
        writer.writeheader()
        writer.writerows(all_laptops)

    print(f"Scraped {len(all_laptops)} laptops → laptops.csv")


if __name__ == "__main__":
    main()

Load & Query in PostgreSQL

Dump the CSV into a relational database, then generate two result tables: (1) each laptop with parsed hardware fields extracted from the raw spec string, and (2) average price grouped by screen size.

Step 1 — Create table & load data
task2_sql.sql — load 
CREATE TABLE IF NOT EXISTS laptops (
    name          TEXT,
    price         NUMERIC(10, 2),
    specification TEXT
);

-- Run inside psql:
\copy laptops
  FROM 'laptops.csv'
  WITH (FORMAT CSV, HEADER TRUE);
How I dumped the data
Method
  • Created the laptops table in PostgreSQL with matching column types
  • Used \copy (client-side) to load laptops.csv — works without superuser privileges
  • FORMAT CSV, HEADER TRUE skips the header row automatically
  • All 117 rows loaded in one command, no data cleaning needed
Step 2 — Table 1: Brand, Spec, HDD, Screen_Size, Price
task2_sql.sql — query 1 
SELECT
    name AS brand,
    specification AS spec,

    -- Storage (HDD/SSD): number after the RAM field
    -- e.g. "4GB, 500GB" → 500, "8GB, 1TB" → 1000
    COALESCE(
        CAST(
            SUBSTRING(specification FROM '\d+GB,\s*(\d+)GB')
            AS INTEGER
        ),
        CAST(
            SUBSTRING(specification FROM '\d+GB,\s*(\d+)TB')
            AS INTEGER
        ) * 1000
    ) AS hdd,

    -- Screen size: leading number before the inch symbol
    CAST(
        SUBSTRING(specification FROM '^([0-9]+\.?[0-9]*)"')
        AS NUMERIC(4, 1)
    ) AS screen_size,

    price

FROM laptops
ORDER BY name;

Step 3 — Table 2: Screen_Size, Avg_Price
task2_sql.sql — query 2 
SELECT
    CAST(
        SUBSTRING(specification FROM '^([0-9]+\.?[0-9]*)"')
        AS NUMERIC(4, 1)
    )                        AS screen_size,

    ROUND(AVG(price), 2)     AS avg_price

FROM laptops
WHERE specification ~ '^[0-9]+"'  -- rows with valid screen size
GROUP BY screen_size
ORDER BY screen_size;

Regular Expressions

Email Validation Pattern

Write a regex pattern that passes or rejects a set of email addresses based on implicit rules derived purely from the test cases — no explicit specification given.

Derived Rules
  • Local part must start with a letter — digits at the start are rejected
  • Allows letters, digits, and underscores in the local part
  • No hyphens allowed anywhere in the local part
  • At most one dot — two or more dots are rejected
  • Dot can be followed by letters, digits, or underscores
Final Pattern
Python · re module ^[a-zA-Z][a-zA-Z0-9_]*(\.[a-zA-Z0-9_]+)?@[a-zA-Z0-9]+\.[a-zA-Z]{2,}$
task3_regex.py
task3_regex.py 
import re

PATTERN = (
    r"^[a-zA-Z][a-zA-Z0-9_]*"
    r"(\.[a-zA-Z0-9_]+)?"
    r"@[a-zA-Z0-9]+\.[a-zA-Z]{2,}$"
)

test_cases = [
    ("my-name@someemail.com",        "NOT PASS"),
    ("myname@someemail.com",         "PASS"),
    ("my.name@someemail.com",        "PASS"),
    ("my.name2019@someemail.com",    "PASS"),
    ("my.name.2019@someemail.com",   "NOT PASS"),
    ("somename.201903@someemail.com","PASS"),
    ("my_name.201903@someemail.com", "PASS"),
    ("201903myname@someemail.com",   "NOT PASS"),
    ("201903.myname@someemail.com",  "NOT PASS"),
]

for email, expected in test_cases:
    got = "PASS" if re.match(PATTERN, email) else "NOT PASS"
    ok  = "✓" if got == expected else "✗"
    print(f"  {ok}  {email:<35}  {got}")
